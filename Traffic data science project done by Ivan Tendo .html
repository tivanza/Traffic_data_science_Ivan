import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv('traffic_data.csv')

df['date_time'] = pd.to_datetime(df['date_time'])

df['month_in_words'] = df['date_time'].dt.strftime('%B')

grouped_data = df.groupby('month_in_words')['traffic_volume'].sum().reset_index()

plt.figure(figsize=(10, 6))
plt.bar(grouped_data['month_in_words'], grouped_data['traffic_volume'], color='blue')
plt.xlabel('Month')
plt.ylabel('Total Traffic Volume')
plt.title('Total Traffic Volume by Month')
plt.xticks(rotation=45)

plt.tight_layout()
plt.show()
